---
title: "Update Badge Collection Metadata"
sidebarTitle: "Update Collection Metadata"
api: "POST https://api.station.express/api/badges/collection/metadata"
description: "Update metadata for an Badge NFT collection."
---

### Body

<ParamField body="contractAddress" type="string" required>
  The token collection smart contract address to edit.
</ParamField>

<ParamField body="metadata" type="object">
    Metadata object to apply to each token in the collection.

    <Expandable title="NFT Metadata">

    <ParamField body="name" type="string" required>
      Name for the token.
    </ParamField>

    <ParamField body="description" type="string" optional>
      Description for the token.
    </ParamField>

    <ParamField body="mediaUrl" type="string" required>
      Url for the media content associated with the token.
    </ParamField>

    <ParamField body="externalUrl" type="string" optional>
      Url for the external link associated with the token.
    </ParamField>

    <ParamField body="traits" type="Trait[]" optional>
        Array of trait objects.

        <Expandable title="Trait">
            <ParamField body="name" type="string" required>
                Name for the trait.
            </ParamField>

            <ParamField body="value" type="string" required>
                Value for the trait.
            </ParamField>
        </Expandable>

    </ParamField>

    </Expandable>

</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="transactionHash" type="string">
  Transaction hash for the newly created token.
</ResponseField>

<ResponseField name="transactionLink" type="string">
  Transaction link for the newly created token, wraps `transactionHash` with the
  appropriate URL for convenience.
</ResponseField>

<ResponseField name="contractAddress" type="string">
  The token collection smart contract address that was edited.
</ResponseField>

<ResponseField name="metadata" type="object">
    Metadata object applied to each token in the collection.

    <Expandable title="NFT Metadata">

    <ResponseField name="name" type="string">
      Name for the token.
    </ResponseField>

    <ResponseField name="description" type="string">
      Description for the token.
    </ResponseField>

    <ResponseField name="mediaUrl" type="string">
      Url for the media content associated with the token.
    </ResponseField>

    <ResponseField name="externalUrl" type="string">
      Url for the external link associated with the token.
    </ResponseField>

    <ResponseField name="traits" type="Trait[]">
        Array of trait objects.

        <Expandable title="Trait">
            <ResponseField name="name" type="string">
                Name for the trait.
            </ResponseField>

            <ResponseField name="value" type="string">
                Value for the trait.
            </ResponseField>
        </Expandable>

    </ResponseField>

    </Expandable>

</ResponseField>
