---
title: "Update an NFT's Unique Metadata"
sidebarTitle: "Update Unique Metadata"
api: "POST https://api.station.express/api/erc721/uniqueMetadata"
description: "Update unique metadata for a specific NFT."
---

### Body

<ParamField body="contractAddress" type="string" required>
  The NFT smart contract address to edit.
</ParamField>

<ParamField body="tokenId" type="string" required>
  The NFT smart contract address to edit.
</ParamField>

<ParamField body="uniqueMetadata" type="object" required>
    Unique metadata to associate with this NFT, which overrides any defaults set at the collection-level.

    <Expandable title="NFT Metadata">

      <ParamField body="name" type="string" optional>
        Name for the NFT.
      </ParamField>

      <ParamField body="description" type="string" optional>
        Description for the NFT.
      </ParamField>

      <ParamField body="mediaUrl" type="string" optional>
        Url for the media content associated with the NFT.
      </ParamField>

      <ParamField body="externalUrl" type="string" optional>
        Url for the external link associated with the NFT.
      </ParamField>

      <ParamField body="traits" type="Trait[]" optional>
        Array of trait objects.

        <Expandable title="Trait">
          <ParamField body="name" type="string" required>
            Name for the trait.
          </ParamField>

          <ParamField body="value" type="string" required>
            Value for the trait.
          </ParamField>
        </Expandable>

      </ParamField>

    </Expandable>

</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="transactionHash" type="string">
  Transaction hash for the newly created token.
</ResponseField>

<ResponseField name="transactionLink" type="string">
  Transaction link for the newly created token, wraps `transactionHash` with the
  appropriate URL for convenience.
</ResponseField>
