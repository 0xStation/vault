---
title: "Create Token"
api: "POST https://api.station.express/api/token/create"
description: "This endpoint creates a new token"
---

### Body

<ParamField body="type" type="string" required>
  The type of token you would like to create. Must be one of `ERC20`, `ERC721`,
  or `ERC1155`
</ParamField>

<ParamField body="name" type="string" required>
  The name of the token. This cannot be changed after creation.
</ParamField>

<ParamField body="symbol" type="string" required>
  The symbol of the token. This cannot be changed after creation.
</ParamField>

<ParamField body="nftMetadataPerToken" type="boolean" optional>
  If this token is an NFT, `true` if metadata will be unique per token or
  `false` if the same metadata applies to all tokens in this collection.
  Defaults to `true`.
</ParamField>

<ParamField body="nftMetadata" type="object" optional>
    Metadata details if token is an NFT and `nftMetadataPerToken` is `false`.

    <Expandable title="NFT Metadata">

<ParamField body="name" type="string" required>
  Name for the token.
</ParamField>

<ParamField body="description" type="string" optional>
  Description for the token.
</ParamField>

<ParamField body="mediaUrl" type="string" required>
  Url for the media content associated with the NFT
</ParamField>

<ParamField body="externalUrl" type="string" optional>
  Url for the external link associated with the NFT.
</ParamField>

<ParamField body="traits" type="Trait[]" optional>
    Array of trait objects.

    <Expandable title="Trait">
        <ParamField body="name" type="string" required>
            Name for the trait.
        </ParamField>

        <ParamField body="value" type="string" required>
            Value for the trait.
        </ParamField>
    </Expandable>

</ParamField>

</Expandable>

</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="transactionHash" type="string">
  Transaction hash for the newly created token.
</ResponseField>

<ResponseField name="transactionLink" type="string">
  Transaction link for the newly created token, wraps `transactionHash` with the
  appropriate URL for convenience.
</ResponseField>

<ResponseField name="token" type="object">

The newly created token smart contract.

<Expandable title="Token">

<ResponseField name="address" type="string">
  The Ethereum address this token contract is deployed at.
</ResponseField>

<ParamField body="type" type="string">
  The type of token, one of `ERC20`, `ERC721`, or `ERC1155`.
</ParamField>

<ResponseField name="name" type="string">
  The name of the token. This cannot be changed after creation.
</ResponseField>

<ResponseField name="symbol" type="string">

    <ResponseField name="traits" type="Trait[]">
