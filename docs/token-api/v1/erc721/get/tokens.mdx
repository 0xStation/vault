---
title: "Fetch Membership NFTs from the Collection"
sidebarTitle: "Fetch NFTs"
api: "GET https://members.station.express/api/v1/erc721/tokens"
description: "Read all NFT data including current owner and history."
---

### Query Params

<ParamField body="chainId" type="number" required>
  The id of the chain supporting the NFT. This cannot be changed after creation.
</ParamField>

<ParamField query="contractAddress" type="string" required>
  The NFT smart contract address.
</ParamField>

### Response

<ResponseField name="success" type="boolean">
  Indicates whether the call was successful.
</ResponseField>

<ResponseField name="nfts" type="Nft[]">
  Array of NFT objects.

  <Expandable title="NFT">

      <ResponseField name="holderAddress" type="string">
        Address of the current holder.
      </ResponseField>

      <ResponseField name="holderSince" type="number">
        UNIX Timestamp when the current holder became this token's holder.
      </ResponseField>

      <ResponseField name="tokenId" type="string">
        Token id of the NFT.
      </ResponseField>

      <ResponseField name="name" type="string">
        Name of the NFT.
      </ResponseField>

      <ResponseField name="description" type="string">
        Description of the NFT.
      </ResponseField>

      <ResponseField name="mediaUrl" type="string">
        Url of the media content associated with the NFT.
      </ResponseField>

      <ResponseField name="externalUrl" type="string">
        Url of the external link associated with the NFT.
      </ResponseField>

      <ResponseField name="traits" type="Trait[]">
        Array of trait objects.

        <Expandable title="Trait">
          <ResponseField name="name" type="string">
            Name for the trait.
          </ResponseField>

          <ResponseField name="value" type="string">
            Value for the trait.
          </ResponseField>
        </Expandable>

      </ResponseField>

    </Expandable>

</ResponseField>

